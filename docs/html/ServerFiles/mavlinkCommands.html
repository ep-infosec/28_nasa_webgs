import time
from pymavlink import mavutil, mavwp
import struct
import logging

timeout = .5

# Mavlink Commands


def requestDataStream(BAUD, master):
    master.mav.request_data_stream_send(master.target_system,
                                        master.target_component,
                                        mavutil.mavlink.MAV_DATA_STREAM_ALL,
                                        BAUD,
                                        1)


def recvAndLog(master, mlog, forwarding, traffic=0):
    if master != 'NONE':
        if traffic == 0:
            master.pre_message()
            m = master.recv()
            if len(m) > 0:

                if master.first_byte:
                    master.auto_mavlink_version(m)

                msg = master.mav.parse_char(m)

                # mlog.write(m)          use this to generate a tlog.raw
                master.post_message(msg)
                usec = int(time.time() * 1.0e6)
                usec = (usec & ~3 | 3)
                x = bytearray(struct.pack('>Q', usec)) + m
                mlog.write(x)
                if forwarding is not None:
                    forwarding.write(x)

                # good for debuging
                # (tusec,) = struct.unpack('>Q', x[0:8])
                # print(usec, tusec, len(x), len(m), len(x)-len(m), m,  x)

                return msg
        else:
            timestamp = int(time.time() * 1.0e6)
            timestamp = (timestamp & ~3 | 3)

            # convert traffic to adsb message
            t = traffic
            name = str(t.name)
            if len(name) < 9:
                x = 9 - len(name)
                name = ' '*x + name
            else:
                name = name[:9]

            m = master.mav.adsb_vehicle_encode(
                int(t.name), int(t.x*1e7), int(t.y*1e7), int(0), int(t.z*1000), int(t.H*100), int(t.S*100), int(0), name.encode('ascii'), int(t.emiter), int(1), int(64), int(0))

            # convert traffic bytearray
            pm = m.pack(master.mav)

            x = bytearray(struct.pack('>Q', timestamp)) + pm
            mlog.write(x)
            if forwarding is not None:
                forwarding.write(x)

            # for debugging
            # dm = master.mav.parse_char(pm)
            # print(dm)
            return traffic
    else:
        return None


def recvMatchAndLog(master, mlog, forwarding, t):
    logger = logging.getLogger()
    t_now = time.time()
    timeout = 2
    while True:
        msg = recvAndLog(master, mlog, forwarding)
        if msg is not None:
            match = [x for x in t if x in str(msg)]
            if len(match) > 0:
                logger.info('recv, match, and log {}'.format(msg))
                return msg
        elif time.time() - t_now < timeout:
            time.sleep(.01)
        elif time.time() - t_now > timeout:
            return


def sendHeartbeat(master):
    master.mav.command_long_send(master.target_system,
                                 master.target_component,
                                 mavutil.mavlink.HEARTBEAT,
                                 0,  # confirmation
                                 6,  # type
                                 8,  # autopilot
                                 0,  # base mode
                                 0,  # custom mode
                                 0,  # status
                                 3,  # version
                                 0
                                 )


def requestCount(master, mlog, forwarding):
    master.mav.mission_request_list_send(master.target_system,
                                         master.target_component,
                                         mavutil.mavlink.MAV_MISSION_TYPE_MISSION)

    msg_in = recvMatchAndLog(master, mlog, forwarding, ['MISSION_COUNT'])
    return msg_in

def setFlightMode(mode, master, mlog, forwarding):
    logger = logging.getLogger()
    mode_id = master.mode_mapping()[mode]
    logger.info('SERVER: Mode Id {}'.format(mode_id))
    master.mav.set_mode_send(
        master.target_system,
        mavutil.mavlink.MAV_MODE_FLAG_CUSTOM_MODE_ENABLED,
        mode_id)

    msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
    logger.info('SERVER: Set Mode: {}'.format(msg_in))

    return msg_in

def startFlight(icarous_flag, master, mlog, forwarding):
    logger = logging.getLogger()
    if icarous_flag == '1':
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_MISSION_START,
                                     0, 0, 0, 0, 0, 0, 0, 0)

        return

    else:
        # change mode to guided
        mode = master.mode_mapping()['GUIDED']
        master.mav.set_mode_send(
            master.target_system,
            mavutil.mavlink.MAV_MODE_FLAG_CUSTOM_MODE_ENABLED,
            mode)

        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))

        # arm the quad
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_COMPONENT_ARM_DISARM,
                                     0,
                                     1, 0, 0, 0, 0, 0, 0)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))
        time.sleep(1)

        # Throttle up
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_DO_CHANGE_SPEED,
                                     0,
                                     0, 10, 10, 1, 0, 0, 0)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))
        time.sleep(1)

        # Takeoff
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_NAV_TAKEOFF,
                                     0,
                                     0, 0, 0, 0, 37.0866015, -76.3788989, 15)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))
        time.sleep(1)

        # change mode to Auto
        mode = master.mode_mapping()['AUTO']
        master.mav.set_mode_send(
            master.target_system,
            mavutil.mavlink.MAV_MODE_FLAG_CUSTOM_MODE_ENABLED,
            mode)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        return str(msg_in)

    if icarous_flag == '1':
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_MISSION_START,
                                     0, 0, 0, 0, 0, 0, 0, 0)

        return

    else:
        # change mode to guided
        mode = master.mode_mapping()['GUIDED']
        master.mav.set_mode_send(
            master.target_system,
            mavutil.mavlink.MAV_MODE_FLAG_CUSTOM_MODE_ENABLED,
            mode)

        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))

        # arm the quad
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_COMPONENT_ARM_DISARM,
                                     0,
                                     1, 0, 0, 0, 0, 0, 0)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))
        time.sleep(1)

        # Throttle up
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_DO_CHANGE_SPEED,
                                     0,
                                     0, 10, 10, 1, 0, 0, 0)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))
        time.sleep(1)

        # Takeoff
        master.mav.command_long_send(master.target_system, master.target_component,
                                     mavutil.mavlink.MAV_CMD_NAV_TAKEOFF,
                                     0,
                                     0, 0, 0, 0, 37.0866015, -76.3788989, 15)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        logger.info('SERVER: message in {}'.format(msg_in))
        time.sleep(1)

        # change mode to Auto
        mode = master.mode_mapping()['AUTO']
        master.mav.set_mode_send(
            master.target_system,
            mavutil.mavlink.MAV_MODE_FLAG_CUSTOM_MODE_ENABLED,
            mode)
        msg_in = recvMatchAndLog(master, mlog, forwarding, ['COMMAND_ACK'])
        return str(msg_in)
